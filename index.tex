\newcommand{\draw}[2]{
	\begin{tikzpicture}
	\draw[borde] (0,0) -- (0.5*#1,0) -- (0.5*#1,0.5*#2) -- (0,0.5*#2) -- (0,0);
	\foreach \y in {1,...,#2} {
		\ifnum\y<#2%
			\draw[xaxis] (0,\y*0.5) -- (0.5*#1,\y*0.5);%
		\fi%
	}
	\foreach \x in {1,...,#1} {
		\ifnum\x<#1%
			\draw[yaxis] (\x*0.5,0) -- (\x*0.5,0.5*#2);%
		\fi%
	}
	\end{tikzpicture}
}
\newcommand{\main}[4]{

		\draw{12}{#2}\hfill\draw{#1}{#2}
	\fi%
	\ifnum #4=1%
		\draw{#1}{#2}\hfill\draw{12}{#2}
	\fi%
\end{flushleft}

	\draw{\theexample}{10}
\end{center}
}

}
